
# Relação de Comandos para o Git


 - cd .. VOLTA DIRETÓRIO
 - cd diretórioX VAI PARA O DIRETÓRIOX
 - mv nomearquivo ./receitas/
 - mkdir CRIA DIRETÓRIO
 - ls LISTAR
 - ls -a LISTAR INCLUINDO OCULTOS
 - git init
 - git clone httpslink CLONA DIRETÓRIO PÚBLICO DO LINK
 - git remote -v MOSTRA PRA ONDE ESTÁ APONTADO O DIRETÓRIO
 - git status MOSTRA O STATUS DOS ARQUIVOS
 - git add * PASSAR OS ARQUIVOS PARA STAGED
 - git commit -m "Comentário" COMMITA ALTERAÇÕES
 - git remote add origin httpRepoGithub LINKA REPO NOVO DO GH COM LOCAL

 - git push origin master ENVIA ALTERAÇÕES PARA O GITHUB
 - git pull origin master PUXA ALTERAÇÕES DO GITHUB
